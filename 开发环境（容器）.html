Alpine 一些初始化

bash
# 国内镜像源，修改为阿里的源
sed -i 's/dl-cdn.alpinelinux.org/mirrors.aliyun.com/g' /etc/apk/repositories

# npm淘宝镜像
npm config set registry https://registry.npmmirror.com

# 更新
apk update

# 安装telnet
apk add busybox-extras

# 安装git
apk add git

# 安装pnpm 
npm install -g pnpm

# 如果无效添加连接
ln -s /usr/local/node/bin/pnpm /usr/local/bin/

# go国内源
go env -w GOPROXY=https://goproxy.cn,direct

git config --global user.email "moujunmore@qq.com"
git config --global user.name "某君more"
