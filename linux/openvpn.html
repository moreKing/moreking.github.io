1. 生成证书脚本 ​

bash
#!/bin/bash
days=36500

create_ca(){
   [ -f ca.key ] || openssl genrsa -out ca.key 2048
   [ -f ca.crt ] || openssl req -x509 -new -nodes -key ca.key -subj "/CN=ca" -days ${days} -out ca.crt
}

create_ssl(){

   echo -n "请输入 SSL 证书名称："
   read ssl_name
   cat > openssl.conf << EOF
[req]
default_bits = 2048
prompt = no
default_md = sha256
req_extensions = v3_req
distinguished_name = dn

[ dn ]
C = CN
ST = AH
L = CZ
O = moujun
OU = it

[ v3_req ]
keyUsage=keyEncipherment,dataEncipherment
extendedKeyUsage=serverAuth,clientAuth

EOF

# 创建 SSL 证书
   [ -f ${ssl_name}.key ] || openssl genrsa -out ${ssl_name}.key 2048
   [ -f ${ssl_name}.csr ] || openssl req -new -key ${ssl_name}.key -subj "/CN=${ssl_name}" -out ${ssl_name}.csr -config openssl.conf
   [ -f ${ssl_name}.crt ] || openssl x509 -req -sha256 -in ${ssl_name}.csr -CA ca.crt -CAkey ca.key -CAcreateserial -out ${ssl_name}.crt -days ${days} -extensions v3_req  -extfile openssl.conf -sha256
   rm -f openssl.conf

   echo "
证书生成成功，有效期为100年！
   "
}

echo "欢迎使用SSL证书生成脚本！"
    
while true 
do
    printf "\n\n主菜单：
    1. 生成 CA  证书
    2. 生成 SSL 证书
    3. 生成 DH  密钥
    4. 生成 tls-auth
    0. 其他任意键退出"
    printf  "\n\n请输入需要操作的选项："
    read name

    if [ -z "$name" ];then
    continue
    fi

    if [ ${name} = "1" ]
    then
    create_ca

    elif [ ${name} = "2" ]
    then

    create_ssl

    elif [ ${name} = "3" ]
    then

    openssl dhparam -out dh.pem 2048

    elif [ ${name} = "4" ]
    then

    openvpn --genkey --secret ta.key

    else
    echo "退出"
    exit 0
    fi

done

2. 创建服务器端配置文件 ​

conf
port 1194 #端口
proto udp #协议可以改成tcp协议
dev tun #采用路由隧道模式tun
ca /etc/openvpn/ca.crt #ca证书文件位置
cert /etc/openvpn/server.crt #服务端公钥名称
key /etc/openvpn/server.key #服务端私钥名称
dh /etc/openvpn/dh.pem #交换证书
server 10.10.10.0 255.255.255.0 #给客户端分配地址池，注意：不能和VPN服务器内网网段有相同
push "route 172.18.63.0 255.255.0.0" #允许客户端访问内网172.18.63.195网段
# push "redirect-gateway def1 bypass-dhcp"  #将所有客户端流量通过 VPN 服务器进行转发，常用于将客户端的默认路由指向 VPN 服务器
# push "dhcp-option DNS 8.8.8.8"            #设置客户端DNS
# ifconfig-pool-persist ipp.txt #地址池记录文件位置
cipher AES-256-CBC
keepalive 10 120 #存活时间，10秒ping一次,120 如未收到响应则视为断线
max-clients 100 #最多允许100个客户端连接
status openvpn-status.log #日志记录位置
verb 3 #设置日志详细程度，level 为 0 到 11 的整数，level 越高，输出越详细
# client-to-client #客户端与客户端之间支持通信
# log /var/log/openvpn.log #openvpn日志记录位置
tls-auth /etc/openvpn/ta.key 0 #防止Ddos攻击 
persist-key #通过keepalive检测超时后，重新启动VPN，不重新读取keys，保留第一次使用的keys。
persist-tun #检测超时后，重新启动VPN，一直保持tun是linkup的。否则网络会先linkdown然后再linkup
# duplicate-cn # 一个用户多人同时登录

# 优化网速
# 确保 TUN/TAP 网络接口在会话重启后保持不变，不会被重置
persist-key
# 确保密钥文件在会话重启后保持不变
persist-tun

#comp-lzo            #取消传输数据压缩
sndbuf 393216
rcvbuf 393216
push "sndbuf 393216"
push "rcvbuf 393216"


script-security 3
auth-user-pass-verify /etc/openvpn/auth.py via-env    #指定用户认证脚本
username-as-common-name
verify-client-cert none

使用账号密码方式登录 ​

密码验证脚本

python
#!/usr/bin/python3.6
# -*- coding: utf-8 -*-

# @auther : more
import os
from datetime import datetime


# 获取环境变量
username = os.getenv("username")
password = os.getenv("password")
client_ip = os.getenv("client_ip")

# 获取客户端的ip和端口
untrusted_ip= os.getenv("untrusted_ip")
untrusted_port = os.getenv("untrusted_port")


logFile = "/etc/openvpn/password-auth.log"

def write_log(status):
    # 打开文件用于追加写入
    with open(logFile, 'a', encoding='utf-8') as file:
        
        file.write("时间: {}  客户端IP: {} 客户端端口: {}  ".format(datetime.now().strftime("%Y-%m-%d %H:%M:%S"), untrusted_ip, untrusted_port))

        # 追加多行内容
        if status:
            file.write("账号：{} 密码: {}  clientIP {} 登录成功\n".format(username, password, client_ip))
        else:
            file.write("账号：{} 密码: {}  clientIP {} 登录失败\n".format(username, password, client_ip))


if username == "test" and password == "test":
    write_log(True)
    exit(0)
else:
    write_log(False)
    exit(1)

客户端配置文件 client.ovpn

conf
client
dev tun
proto udp
remote 47.122.0.104 1194
resolv-retry infinite
nobind
persist-key
persist-tun
tls-auth ta.key 1
remote-cert-tls server
ca ca.crt                   #该文件一定要与服务器端ca.crt是同一个文件
;cert client.crt            #定义客户端的证书文件，此处注释即可
;key client.key             #定义客户端的密钥文件,此处不需要开启   
;comp-lzo                    #启用允许数据压缩，和Server端保持一致
verb 3
cipher AES-256-CBC
auth-user-pass              #使用用户名密码登录openvpn服务器
