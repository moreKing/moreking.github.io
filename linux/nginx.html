conf
server {
    listen 443 ssl;  # 监听443端口，启用SSL
    server_name aca.moujun.top;  # 二级域名

    ssl_certificate /etc/nginx/cert/cert.pem;  # 替换为你的证书文件路径
    ssl_certificate_key /etc/nginx/cert/private.key;  # 替换为你的私钥文件路径

    location / {
        proxy_pass https://192.168.2.31:7443;  # 后端服务地址
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header Upgrade $http_upgrade; # 处理websocket
        proxy_set_header Connection 'upgrade'; # 处理websocket
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
